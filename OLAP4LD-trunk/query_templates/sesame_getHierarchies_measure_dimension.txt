?CATALOG_NAME ?SCHEMA_NAME ?CUBE_NAME ?DIMENSION_UNIQUE_NAME ?HIERARCHY_UNIQUE_NAME ?HIERARCHY_NAME ?HIERARCHY_CAPTION ?DESCRIPTION 
{{{STANDARDFROM}}} 
WHERE { 
?CUBE_NAME a qb:DataSet.

BIND('Measures' as ?HIERARCHY_NAME).

} ORDER BY ?CUBE_NAME
